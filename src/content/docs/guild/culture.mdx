import { getCollection, render } from 'astro:content';
import { sortByTitle } from '@lib/sortFunctions';
import IconTabs from '@components/common/IconTabs.astro';

export const memes = sortByTitle(await getCollection('memes'));
export const entries = await Promise.all(
	memes.map(async (m) => ({
		slug: m.id,
		title: m.data.title,
		icon: m.data.icon,
		subtitle: `Article written by ${m.data.author}`,
		Content: (await render(m)).Content,
	}))
);

Over the years, VETS has accumulated a unique culture of inside jokes and guild-specific memes: here are overviews of some.

<IconTabs hashPrefix="meme" entries={entries} />

<script is:inline>{`(function() {
  var sEl = document.createElement("style");
  sEl.textContent = "#bt-input{display:block;width:100%;background:rgba(0,0,0,0.15);caret-color:currentColor;color:inherit;font-family:Inter,system-ui,-apple-system,sans-serif;font-size:1rem;line-height:1.625;padding:0.75rem;border:2px solid rgba(255,255,255,0.2);border-radius:6px;resize:none;min-height:10rem;outline:none;margin:0;margin-top:1rem;box-sizing:border-box}#bt-input:focus{border-color:rgba(255,255,255,0.4)}#bt-input::placeholder{color:rgba(255,255,255,0.3)}#bt-bar{display:flex;justify-content:space-between;align-items:center;margin-top:0.5rem;padding:0 0.25rem;font-size:0.875rem}#bt-counter{opacity:0.6}#bt-copy{margin-top:0.75rem;padding:0.5rem 1.5rem;border-radius:6px;font-weight:600;cursor:pointer;transition:all 0.2s;border:1px solid rgba(255,255,255,0.3);background:rgba(255,255,255,0.1);color:inherit;font-size:0.95rem}#bt-copy:not(:disabled):hover{background:rgba(255,255,255,0.2)}#bt-copy:disabled{cursor:not-allowed;opacity:0.4}";
  document.head.appendChild(sEl);

  var inp = document.getElementById("bt-input");
  var ctr = document.getElementById("bt-counter");
  var st = document.getElementById("bt-status");
  var cp = document.getElementById("bt-copy");
  if (!inp || !ctr || !st || !cp) return;

  inp.addEventListener("input", function() {
    ctr.textContent = inp.value.length + " / 256";
    if (inp.value.length > 0) {
      st.textContent = "Ready to copy!";
      st.style.color = "#4ade80";
      cp.disabled = false;
      cp.style.opacity = "1";
    } else {
      st.textContent = "";
      st.style.color = "";
      cp.disabled = true;
      cp.style.opacity = "0.4";
    }
  });

  cp.addEventListener("click", function() {
    if (!cp.disabled && navigator.clipboard) {
      navigator.clipboard.writeText(inp.value).then(function() {
        var orig = cp.textContent;
        cp.textContent = "Copied!";
        setTimeout(function() { cp.textContent = orig; }, 2000);
      });
    }
  });
})();`}</script>
